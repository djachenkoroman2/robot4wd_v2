#include <Arduino.h>

const byte interruptPin = 2;
volatile unsigned int rot = 0; // volatile для переменных в прерываниях
volatile unsigned long lastTime = 0;
unsigned long speedRPM = 0;
unsigned long lastCalcTime = 0;

void detect() {
    unsigned long currentTime = millis();
    
    // Антидребезг - игнорируем срабатывания чаще чем раз в 10 мс
    // if (currentTime - lastTime > 10) {
        rot++;
    //     lastTime = currentTime;
    // }
}

void setup() {
    Serial.begin(9600);
    
    pinMode(interruptPin, INPUT_PULLUP);
    attachInterrupt(digitalPinToInterrupt(interruptPin), detect, RISING);
    
    lastTime = millis();
    lastCalcTime = millis();
}

void loop() {
    unsigned long currentTime = millis();
    
    // Вычисляем скорость каждые 100 мс
    if (currentTime - lastCalcTime >= 100) {
        // Отключаем прерывания на время чтения/сброса переменной
        noInterrupts();
        unsigned int rotations = rot;
        rot = 0;
        interrupts();
        
        // Вычисляем RPM
        if (lastCalcTime > 0) {
            unsigned long timeDiff = currentTime - lastCalcTime;
            if (timeDiff > 0 && rotations > 0) {
                speedRPM = (rotations * 60000UL) / timeDiff;
            } else {
                speedRPM = 0; // Если вращений не было
            }
        }
        
        lastCalcTime = currentTime;
    }
    
    // Вывод значения (можно реже)
    static unsigned long lastPrintTime = 0;
    if (currentTime - lastPrintTime >= 200) {
        Serial.println(speedRPM);
        lastPrintTime = currentTime;
    }
    
    // Не используем delay(), чтобы не блокировать выполнение
    // Можно добавить короткую задержку для стабильности
    //delay(1);
}
